# 引入代码文件

//客户端配置文件
import { defineClientConfig, usePagesData } from "@vuepress/client";

export default defineClientConfig({
  //app实例   router路由  sitedata配置信息
  enhance({ app, router, siteData }) {
    //获取每个页面实例
    const pageData = usePagesData();
    Promise.all(
      Object.keys(pageData.value).map((key) => pageData.value[key]())
    ).then((res) => {
      //全局挂载，每个页面都可以获取到所有页面的实例
      app.provide("pages", res);
    });
  },
  //进入页面调用
  setup() {},
  //注册全局组件
  rootComponents: [],
});
